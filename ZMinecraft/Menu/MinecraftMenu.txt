Class OptionMenuItemMinecraftOption : OptionMenuItemOption
{
	String mTooltip;
	bool mInvertGrayCheck;
	OptionMenuItemMinecraftOption Init(String label, String tooltip, Name command, Name values, CVar graycheck = null, int center = 0, int invertgraycheck = 0)
	{
		mTooltip = tooltip;
		mInvertGrayCheck = invertgraycheck;
		Super.Init(label, command, values, graycheck, center);
		return self;
	}
	
	override bool isGrayed()
	{
		return mGrayCheck && (mInvertGrayCheck ? mGrayCheck.GetInt() : !mGrayCheck.GetInt());
	}
}

Class OptionMenuItemMinecraftSlider : OptionMenuItemSlider
{
	String mTooltip;

	OptionMenuItemMinecraftSlider Init(String label, String tooltip, Name command, double min, double max, double step, int showval = 1)
	{
		mTooltip = tooltip;
		Super.Init(label, command, min, max, step, showval);
		return self;
	}
}

Class OptionMenuItemMinecraftControl : OptionMenuItemControl
{
	String mTooltip;

	OptionMenuItemMinecraftControl Init(String label, String tooltip, Name command)
	{
		mTooltip = tooltip;
		Super.Init(label, command);
		return self;
	}
}

Class MinecraftMenu : OptionMenu
{
	override void Drawer ()
	{
		Super.Drawer();

		String tt = "";

		if (mDesc.mSelectedItem > 0)
		{
			if (mDesc.mItems[mDesc.mSelectedItem] is "OptionMenuItemMinecraftOption")
			{
				tt = StringTable.Localize(OptionMenuItemMinecraftOption(mDesc.mItems[mDesc.mSelectedItem]).mTooltip);
			}

			if (mDesc.mItems[mDesc.mSelectedItem] is "OptionMenuItemMinecraftSlider")
			{
				tt = StringTable.Localize(OptionMenuItemMinecraftSlider(mDesc.mItems[mDesc.mSelectedItem]).mTooltip);
			}

			if (mDesc.mItems[mDesc.mSelectedItem] is "OptionMenuItemMinecraftControl")
			{
				tt = StringTable.Localize(OptionMenuItemMinecraftControl(mDesc.mItems[mDesc.mSelectedItem]).mTooltip);
			}

			/*
			// [Nash] This doesn't work because mUh eXpLoiTs lmaaaaoooo
			if (mDesc.mItems[mDesc.mSelectedItem] is "OptionMenuItemMinecraftSafeCommand")
			{
				tt = StringTable.Localize(OptionMenuItemMinecraftSafeCommand(mDesc.mItems[mDesc.mSelectedItem]).mTooltip);
			}
			*/
		}

		if (tt.Length() > 0)
		{
			Screen.DrawText (OptionFont(), OptionMenuSettings.mFontColorValue,
			//	(Screen.GetWidth() - SmallFont.StringWidth (tt) * CleanXfac_1) / 2,
				((Screen.GetWidth() / 2) - ((SmallFont.StringWidth (tt) * CleanXfac_1) / 2)),
				BigFont.GetHeight() * CleanYfac_1 * 2.5,
				tt,
				DTA_CleanNoMove_1, true);
		}
	}
}
